#include <stdio.h>
#define MAX 30

int main() {
    int n, nf, i, j, pageFaults = 0, time = 0;
    int ref[MAX], frames[MAX], counter[MAX];

    printf("Enter number of pages in reference string: ");
    scanf("%d", &n);

    printf("Enter reference string: ");
    for(i = 0; i < n; i++)
        scanf("%d", &ref[i]);

    printf("Enter number of frames: ");
    scanf("%d", &nf);

    for(i = 0; i < nf; i++) {
        frames[i] = -1;
        counter[i] = 0;
    }

    printf("\nPage\tFrames\n");

    for(i = 0; i < n; i++) {
        int found = 0;
        time++;

        for(j = 0; j < nf; j++) {
            if(frames[j] == ref[i]) {
                found = 1;
                counter[j] = time;
                break;
            }
        }

        if(!found) {
            pageFaults++;

            int empty = -1;
            for(j = 0; j < nf; j++) {
                if(frames[j] == -1) {
                    empty = j;
                    break;
                }
            }

            if(empty != -1) {
                frames[empty] = ref[i];
                counter[empty] = time;
            } else {
                int lru = 0;
                for(j = 1; j < nf; j++)
                    if(counter[j] < counter[lru])
                        lru = j;
                frames[lru] = ref[i];
                counter[lru] = time;
            }
        }

        printf("%d\t", ref[i]);
        for(j = 0; j < nf; j++) {
            if(frames[j] != -1)
                printf("%d ", frames[j]);
            else
                printf("- ");
        }
        printf("\n");
    }

    printf("\nTotal Page Faults: %d\n", pageFaults);

    return 0;
}
