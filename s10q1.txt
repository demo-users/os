#include <stdio.h>

#define MAX 30

int main() {
    int ref[MAX], frames[MAX], n, nf, i, j, k = 0, pageFaults = 0, found;

    printf("Enter number of pages in reference string: ");
    scanf("%d", &n);

    printf("Enter reference string: ");
    for (i = 0; i < n; i++)
        scanf("%d", &ref[i]);

    printf("Enter number of frames: ");
    scanf("%d", &nf);

    for (i = 0; i < nf; i++)
        frames[i] = -1;

    printf("\nPage\tFrames\t\tPage Fault\n");
    for (i = 0; i < n; i++) {
        found = 0;
        for (j = 0; j < nf; j++) {
            if (frames[j] == ref[i]) {
                found = 1;
                break;
            }
        }

        if (!found) {
            frames[k] = ref[i];
            k = (k + 1) % nf;
            pageFaults++;
        }

        printf("%d\t", ref[i]);
        for (j = 0; j < nf; j++) {
            if (frames[j] != -1)
                printf("%d ", frames[j]);
            else
                printf("- ");
        }

        if (!found)
            printf("\tYes");
        else
            printf("\tNo");
        printf("\n");
    }

    printf("\nTotal Page Faults = %d\n", pageFaults);
    return 0;
}
