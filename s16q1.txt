#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>
#include <ctype.h>

#define MAX 100

void count_chars(char *filename) {
    FILE *file = fopen(filename, "r");
    if (!file) {
        printf("File nahi khul paya: %s\n", filename);
        return;
    }
    int count = 0;
    while (fgetc(file) != EOF) count++;
    fclose(file);
    printf("Number of characters: %d\n", count);
}

void count_words(char *filename) {
    FILE *file = fopen(filename, "r");
    if (!file) {
        printf("File nahi khul paya: %s\n", filename);
        return;
    }
    int count = 0;
    int in_word = 0;
    char ch;
    while ((ch = fgetc(file)) != EOF) {
        if (isspace(ch))
            in_word = 0;
        else if (!in_word) {
            in_word = 1;
            count++;
        }
    }
    fclose(file);
    printf("Number of words: %d\n", count);
}

int main() {
    char input[MAX];
    char *args[10];

    while (1) {
        printf("myshell$ ");
        fflush(stdout);

        if (fgets(input, sizeof(input), stdin) == NULL)
            break;

        input[strcspn(input, "\n")] = 0;

        int i = 0;
        char *token = strtok(input, " ");
        while (token != NULL && i < 10) {
            args[i++] = token;
            token = strtok(NULL, " ");
        }
        args[i] = NULL;

        if (args[0] == NULL)
            continue;

        if (strcmp(args[0], "exit") == 0)
            break;

        if (strcmp(args[0], "count") == 0 && i == 3) {
            if (strcmp(args[1], "c") == 0)
                count_chars(args[2]);
            else if (strcmp(args[1], "w") == 0)
                count_words(args[2]);
            else
                printf("Invalid option. Use 'count c filename' or 'count w filename'.\n");
            continue;
        }

        pid_t pid = fork();
        if (pid == 0) {
            execvp(args[0], args);
            printf("Command chal nahi payi.\n");
            exit(1);
        } else {
            wait(NULL);
        }
    }
    return 0;
}
