#include <stdio.h>

int main() {
    int n, refLen, i, j, k, pageFaults = 0;

    printf("Enter number of pages in reference string: ");
    scanf("%d", &refLen);

    int refStr[50];
    printf("Enter the reference string: ");
    for (i = 0; i < refLen; i++)
        scanf("%d", &refStr[i]);

    printf("Enter number of frames: ");
    scanf("%d", &n);

    int frame[n];
    int lastUsed[n];

    for (i = 0; i < n; i++) {
        frame[i] = -1;
        lastUsed[i] = -1;
    }

    printf("\nPage\tFrames\t\tPage Fault\n");

    for (i = 0; i < refLen; i++) {
        int page = refStr[i];
        int found = 0;

        for (j = 0; j < n; j++) {
            if (frame[j] == page) {
                found = 1;
                lastUsed[j] = i;
                break;
            }
        }

        if (!found) {
            int replaceIndex = -1;

            for (j = 0; j < n; j++) {
                if (frame[j] == -1) {
                    replaceIndex = j;
                    break;
                }
            }

            if (replaceIndex == -1) {
                replaceIndex = 0;
                for (j = 1; j < n; j++) {
                    if (lastUsed[j] > lastUsed[replaceIndex])
                        replaceIndex = j;
                }
            }

            frame[replaceIndex] = page;
            lastUsed[replaceIndex] = i;
            pageFaults++;
        }

        printf("%d\t", page);
        for (j = 0; j < n; j++) {
            if (frame[j] != -1)
                printf("%d ", frame[j]);
            else
                printf("- ");
        }
        printf("\t\t%s\n", found ? "No" : "Yes");
    }

    printf("\nTotal Page Faults = %d\n", pageFaults);
    return 0;
}
