#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

#define MAX 100

void typeline_top(char *n_str, char *filename) {
    int n = atoi(n_str);
    FILE *file = fopen(filename, "r");
    if (file == NULL) {
        printf("File open nahi ho paya.\n");
        return;
    }

    char line[1000];
    int count = 0;

    while (fgets(line, sizeof(line), file) && count < n) {
        printf("%s", line);
        count++;
    }

    fclose(file);
}

int main() {
    char input[MAX];
    char *args[10];

    while (1) {
        printf("myshell$ ");
        fflush(stdout);

        if (fgets(input, sizeof(input), stdin) == NULL) {
            break;
        }

        input[strcspn(input, "\n")] = 0;

        int i = 0;
        char *token = strtok(input, " ");
        while (token != NULL && i < 10) {
            args[i++] = token;
            token = strtok(NULL, " ");
        }
        args[i] = NULL;

        if (args[0] == NULL) {
            continue;
        }

        if (strcmp(args[0], "exit") == 0) {
            break;
        }

        if (strcmp(args[0], "typeline") == 0) {
            if (i == 3 && args[1][0] == '+') {
                typeline_top(args[1] + 1, args[2]);
            } else {
                printf("Usage: typeline +n filename\n");
            }
            continue;
        }

        pid_t pid = fork();
        if (pid == 0) {
            execvp(args[0], args);
            printf("Command chal nahi payi.\n");
            exit(1);
        } else {
            wait(NULL);
        }
    }

    return 0;
}
