#include <stdio.h>
#define MAX 30

int main() {
    int ref[MAX], frames[MAX];
    int n, nf, i, j, k, pageFaults = 0, flag1, flag2, pos, max, future[MAX];

    printf("Enter number of pages in reference string: ");
    scanf("%d", &n);

    printf("Enter reference string: ");
    for(i = 0; i < n; i++)
        scanf("%d", &ref[i]);

    printf("Enter number of frames: ");
    scanf("%d", &nf);

    for(i = 0; i < nf; i++)
        frames[i] = -1;

    printf("\nPage\tFrames\n");

    for(i = 0; i < n; i++) {
        flag1 = flag2 = 0;

        for(j = 0; j < nf; j++) {
            if(frames[j] == ref[i]) {
                flag1 = flag2 = 1;
                break;
            }
        }

        if(flag1 == 0) {
            for(j = 0; j < nf; j++) {
                if(frames[j] == -1) {
                    frames[j] = ref[i];
                    pageFaults++;
                    flag2 = 1;
                    break;
                }
            }
        }

        if(flag2 == 0) {
            for(j = 0; j < nf; j++) {
                future[j] = -1;
                for(k = i + 1; k < n; k++) {
                    if(frames[j] == ref[k]) {
                        future[j] = k;
                        break;
                    }
                }
            }

            pos = -1;
            max = -1;
            for(j = 0; j < nf; j++) {
                if(future[j] == -1) {
                    pos = j;
                    break;
                } else if(future[j] > max) {
                    max = future[j];
                    pos = j;
                }
            }

            frames[pos] = ref[i];
            pageFaults++;
        }

        printf("%d\t", ref[i]);
        for(j = 0; j < nf; j++) {
            if(frames[j] != -1)
                printf("%d ", frames[j]);
            else
                printf("- ");
        }
        printf("\n");
    }

    printf("\nTotal Page Faults: %d\n", pageFaults);
    return 0;
}
